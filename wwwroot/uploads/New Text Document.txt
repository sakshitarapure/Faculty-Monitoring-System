<?php
// URL of the Google News RSS feed
$feed_url = "https://news.google.com/rss?hl=en-US&gl=US&ceid=US:en";

// Load and parse the feed
$xml = simplexml_load_file($feed_url);

if ($xml === false) {
    echo "Error: Cannot load feed.";
    exit;
}

echo "<h2>Google News Feed</h2>";
echo "<ul>";

foreach ($xml->channel->item as $item) {
    $title = htmlspecialchars($item->title);
    $link = htmlspecialchars($item->link);
    $pubDate = date("F j, Y, g:i a", strtotime($item->pubDate));
    
    echo "<li><a href='$link' target='_blank'>$title</a><br><small>$pubDate</small></li>";
}

echo "</ul>";
?>
